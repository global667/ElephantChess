466void MainWindow::Save() {
467 auto saveFile =
468 QFileDialog::getSaveFileName(this, tr("Save file "), "~/untitled",
469 tr("Portable Game Notation ( *.pgn)"));
470
471 if (saveFile.isNull()) {
472 statusBar()->showMessage(tr("Speichere Datei: fehlgeschlagen"));
473 return;
474 }
475 if (!saveFile.contains(".pgn"))
476 saveFile = saveFile + ".pgn";
477 QFile svfile(saveFile);
478 if (!svfile.open(QIODeviceBase::WriteOnly | QIODeviceBase::Text)) {
479 statusBar()->showMessage(tr("Save file: opening error"));
480 return;
481 }
482
483 QTextStream textstream(&svfile);
484
485 textstream << "[Event \"" << loca->text() << "\"]\n";
486 textstream << "[Date \"" << date->text() << "\"]\n";
487 textstream << "[Round \"" << round->text() << "\"]\n";
488 textstream << "[White \"" << opp1->text() << "\"]\n";
489 textstream << "[Black \"" << opp2->text() << "\"]\n";
490 textstream << "[Result \""
491 << "0-1"
492 << "\"]\n\n";
493
494 int i = 0;
495 for (const auto &m: basemodel.moves) {
496 if (++i % 2 == 1)
497 textstream << QString("%1.").arg((i / 2) + 1);
498 textstream << m << " ";
499 }
500
501 textstream << "\n0-1\n\n";
502
503 svfile.close();
504
505 statusBar()->showMessage(tr("Save file: ") + saveFile);
506}
