125void Engine::initializeZobrist() {
126 std::mt19937_64 rng(0); // Zufallsgenerator mit einem festen Seed für Reproduzierbarkeit
127 std::uniform_int_distribution<uint64_t> dist;
128
129 for (int piece = 0; piece < kPieceTypes; ++piece) {
130 for (int position = 0; position < kBoardSize; ++position) {
131 zobristTable[piece][position] = dist(rng); // Zufällige Werte für jede Position und jeden Spielsteintyp
132 }
133 }
134}

136std::uint64_t Engine::hashBoard(const std::vector<std::vector<std::shared_ptr<Piece>>>& board) {
137 std::uint64_t hash = 0;
138 int i = 0;
139 for (const std::vector<std::shared_ptr<Piece>>& row : board) {
140 for (const auto& piece : row) {
141 if (!piece || piece->color == Color::None)
142 continue; // Kein Spielstein auf diesem Feld (leeres Feld
143 const std::uint64_t name = piece->getId();
144 //hash ^= name; // Hier sollte eine bessere Hash-Funktion verwendet werden
145 hash ^= zobristTable[name][i++]; // XOR mit Zufallswert aus Zobrist-Hash-Tabelle
146 hash = hash * 1099511628211ULL; // Ein großer Prime, der oft in Hashing verwendet wird
147 }
148 i = 0;
149 }
150 return hash;
151}

188int Engine::minimax(int depth, bool maximizingPlayer, const std::vector<std::vector<std::shared_ptr<Piece>>> &board, TranspositionTable *tt) {
189 if (appClosing)
190 return 0;
191 nodes++;
192
193 const std::uint64_t hash = hashBoard(board);
194 if (tt->find(hash) != tt->end()) {
195 //const auto hs = tt->at(hash).value;
196 //qDebug() << "TT hit" " hash: " << hash << " size: " << tt->size() << " nodes: " << nodes << "hashed: " << hashed++;
197 return tt->at(hash).value;;
198 }
199
200 const Color color = maximizingPlayer ? Color::Red : Color::Black;
201 auto moves = Board::getAllValidMoves(color, board);
202
203 if (depth == 0) {
204 const int evaluation = evaluatePosition(board);
205 (*tt)[hash] = {depth, evaluation};
206 return evaluation;
207 }
208 if (moves.empty()) {
209 const int evaluation = Board::isCheckmate(color, board) ? (maximizingPlayer ? -INFINITY_SCORE : INFINITY_SCORE) : 0;
210 (*tt)[hash] = {depth, evaluation};
211 return evaluation;
212 }
213
214
215 QVector<QFuture<int>> futures;
216
217 int eval = maximizingPlayer ? INT_MIN : INT_MAX;
218 auto updateEval = [&eval, maximizingPlayer](const int result) {
219 if (maximizingPlayer) {
220 eval = std::max(eval, result);
221 } else {
222 eval = std::min(eval, result);
223 }
224 };
225
226 for (auto& move : moves) {
227 auto future = QtConcurrent::run([this, depth, move, board, tt, maximizingPlayer]() {
228 std::vector<std::vector<std::shared_ptr<Piece>>> temp_board = board; // Erstelle lokale Kopie für den Thread
229 Board::movePiece(move.first, move.second, temp_board);
230 const int result = minimax(depth - 1, !maximizingPlayer, temp_board, tt);
231 Board::undoMove(move.first, move.second, board[move.first.x][move.first.y], temp_board);
232 return result;
233 });
234 futures.append(future);
235 }
236
237 for (auto &future : futures) {
238 future.waitForFinished();
239 updateEval(future.result());
240 }
241
242 (*tt)[hash] = {depth, eval}; // Speichern der Bewertung
243 //emit updateFromThread();
244 return eval;
245}
