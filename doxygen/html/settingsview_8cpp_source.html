100void SettingsView::ChooseBoardStyle()
101{
102 QStringList items;
103 items << tr("Traditionel") << tr("Traditionel/PNG") << tr("Western simplified");
104
105 bool ok;
106 QString item = QInputDialog::getItem(this, tr("Pieces"), tr("Pieces"), items, 0, false, &ok);
107 if (ok && !item.isEmpty()) {
108 if (item == tr("Traditionel")) {
109 basemodel.viewStyleModeVar = BaseModel::viewStyleMode::traditional_native;
110 } else if (item == tr("Traditionel/PNG")) {
111 basemodel.viewStyleModeVar = BaseModel::viewStyleMode::traditional_png;
112 } else if (item == tr("Western simplified")) {
113 basemodel.viewStyleModeVar = BaseModel::viewStyleMode::western_png;
114 }
115 emit boardStyleChanged();
116 }
117}
