43void BoardView::PrepareEuroPiece(QPicture *pix, int row, int col, int h, int w) {
44
45 const auto piece = basemodel.position.board[row][col];
46 // Check for null pointer and empty piece
47 if (!piece || piece->getName() == "")
48 return;
49
50 const QString path = ":/res/" + piece->euroName + "_" + (piece->getColor()== Color::Red ? "r" : "b" )+ "_z.png";
51
52 const QImage image(path);
53 if (image.isNull()) {
54 qDebug() << "Failed to load image: " << path;
55 return;
56 }
57 QPainter painter;
58 painter.begin(pix);
59 pix->setBoundingRect(QRect(0, 0, w, h));
60
61 // Gradient center and dimensions adjustment
62 QPointF center(w / (2.0 * PIECE_SCALE_FACTOR * cutpWidth), h / (2.0 * PIECE_SCALE_FACTOR * cutpHeight));
63 qreal radius = qMin(w, h) / (2.0 * PIECE_SCALE_FACTOR * qMax(cutpWidth, cutpHeight));
64
65 QRadialGradient gradient(center, radius);
66 gradient.setColorAt(0, QColor::fromRgb(255, 255, 255, 255));
67 gradient.setColorAt(1, QColor::fromRgb(222, 91, 16, 255));
68
69 painter.setBrush(gradient);
70
71 // Draw piece circle
72 const auto width = pix->boundingRect().width();
73 const auto height = pix->boundingRect().height();
74 const auto wdth = static_cast<double>(width) / static_cast<double>(cutpWidth) / PIECE_SCALE_FACTOR;
75 const auto hght = (double) height / (double) cutpHeight / PIECE_SCALE_FACTOR;
76 const auto drawRect = QRect(0, 0, round(wdth), round(hght));
77 painter.drawEllipse(drawRect);
78
79 painter.drawImage(drawRect, image);
80 painter.end();
81}

456 const int x = event->pos().x();
457 const int y = event->pos().y();
458 const Point boardPos = CalcBoardCoords({x, y});
459 const Point coord{10 - boardPos.y, boardPos.x - 1}; // Simplified coordinate conversion

461 pressed = true;
462 basemodel.fromHuman = coord;
463
464 if (const Piece piece = *basemodel.position.board[coord.x][coord.y];
465 piece.getName().isEmpty() || piece.getColor() != basemodel.position.players_color) {
466 pressed = false; // Reset if invalid piece or color
467 return;
468 }
469 update(); // Refresh to show selected piece
470 } else {
471 basemodel.toHuman = coord;
472 if (basemodel.toHuman.x == basemodel.fromHuman.x && basemodel.toHuman.y == basemodel.fromHuman.y) {
473 pressed = false;
474 update(); // Refresh to show changes or revert view
475 return;
476 }
477 if (basemodel.position.board[basemodel.toHuman.x][basemodel.toHuman.y]->getColor() == basemodel.position.
478 players_color) {
479 basemodel.fromHuman = basemodel.toHuman; // Move the selected piece to the new square
480 update(); // Refresh to show changes or revert view
481 return;
482 }
483 pressed = false;
484 auto all_moves = Board::getAllValidMoves(basemodel.position.players_color, basemodel.position.board);
485 for (const auto &[from, to]: all_moves) {
486 if (from == basemodel.fromHuman && to == basemodel.toHuman) {

489 basemodel.toHuman = to;
490 update();
491 return;
492 }
493 }
494 // If no valid move was found, reset the state but do not emit the move
495 update(); // Refresh to show changes or revert view
496 }

500// Calculates from mouse to board coordinates
501constexpr float LEFT_MARGIN = 50.0f;
502constexpr float RIGHT_MARGIN = 50.0f;
503constexpr float TOP_MARGIN = 50.0f;
504constexpr float BOTTOM_MARGIN = 100.0f;
505constexpr float BOARD_COL_POINTS = BaseModel::BoardColPoints; // Assuming this is a valid static constant
506constexpr float BOARD_ROW_POINTS = BaseModel::BoardRowPoints; // Assuming this is a valid static constant
507
508Point BoardView::CalcBoardCoords(Point r) const {
509 const float w = this->width();
510 const float h = this->height();
511
512 // Calculate dimensions of each square on the board
513 const float squareColWidth = (w - LEFT_MARGIN - RIGHT_MARGIN) / BOARD_COL_POINTS;
514 const float squareRowHeight = (h - TOP_MARGIN - BOTTOM_MARGIN) / BOARD_ROW_POINTS;
515
516 // Transform cursor position based on game view
517 const float boardCursorX = basemodel.gameView == Color::Red ? r.x : w - r.x;
518 const float boardCursorY = basemodel.gameView == Color::Red ? r.y : h - r.y;
519
520 // Calculate board coordinates from cursor position
521 int col = static_cast<int>(std::floor(boardCursorX / squareColWidth)) + 1;
522 int row = static_cast<int>(std::floor(boardCursorY / squareRowHeight)) + 1;
523
524 return {col, row};
525}
